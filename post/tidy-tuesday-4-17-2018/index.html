Tidy Tuesday 4-17-2018

Apr 19, 2018 14 min read R, Tidy Tuesday

See here for more about tidy tuesday: https://github.com/rfordatascience/tidytuesday

library(tidyverse)
library(plotly)
library(ggrepel)

Read in data, remove punctuation from column names

inputData <- readxl::read_xlsx("data/4-19-18/global_mortality.xlsx")

names(inputData) <- str_trim(
        str_remove_all(names(inputData), "[[:punct:]]")
        )

Mean of each category grouped by country isn’t great, but for a tidy-tuesday it’ll do…

a2 <- inputData %>% group_by(country) %>%  summarize_if(is.numeric,mean)
a <- a2 %>% select(4:34) %>% replace(is.na(.), 0)
principleComponents <- prcomp(a) %>%
                          .$x %>%
                          as_tibble %>%
                          bind_cols(name=a2$country, .)
ggplot(principleComponents) +
    geom_label(aes(x=PC1,y=PC2,label=name)) +
    ggtitle("First and Second Principle Components")

principleComponents2 <- prcomp(a) %>%
                          .$rotation %>%
                          as_tibble %>%
                          bind_cols(name=colnames(a), .)

HIV in Africa and and Cancer in First-World Countries are separable by the first Principle Component

ggplot(principleComponents2) +
    geom_label(aes(x=PC1,y=PC2,label=name)) +
      ggtitle("First and Second Principle Components, Variable Contributions")

Let’s look at one more set

ggplot(principleComponents) +
    geom_label(aes(x=PC3,y=PC4,label=name)) +
    ggtitle("Third and Fourth Principle Components")

Diabetes vs Malaria are disparate when comparing some Pacific Island nations and some African nations

ggplot(principleComponents2) +
    geom_label(aes(x=PC3,y=PC4,label=name)) +
    ggtitle("Third and Fourth Principle Components Variable Contributions")

The same data as presented above but as geom_bar such as in Julia Silge’s PCA tutorial here.

principleComponents2$name <- factor(principleComponents2$name, levels = principleComponents2$name[order(principleComponents2$PC1)])

w <- ggplot(principleComponents2) +
      geom_bar(stat="identity",aes(x=name,y=PC1,fill=name)) +
      theme(legend.position="none") +
      theme(axis.text.x = element_text(angle = 90, hjust = 1,vjust=.4)) +
      theme(axis.title.x=element_blank()) +
      ggtitle("Factor Contributions to Principle Components")

ggplotly(w)
principleComponents$name <- factor(principleComponents$name, levels = principleComponents$name[order(principleComponents$PC1)])

w <- ggplot(principleComponents) +
        geom_bar(stat="identity",aes(x=name,y=PC1,fill=name)) +
        theme(legend.position="none") +
        theme(axis.text.x = element_blank()) +
        xlab("Hover Over Data for Country Names") +   
        ggtitle("Countries Separated in the First Principle Components Dimension")


ggplotly(w)
