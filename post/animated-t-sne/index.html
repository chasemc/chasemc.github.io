Animated t-SNE

Apr 18, 2018 2 min read R, Instructional, ggplot, Rtsne, gganimate, tweenr

Load libraries

library(Rtsne)
library(createdatasets)
library(tweenr)
library(gganimate)
library(ggplot2)

set.seed(42)

Create data for t-SNE

# Get MNIST data for t-SNE
df_mnist    <- createMNIST()
df_sampled  <- df_mnist[sample(1:nrow(df_mnist),1000),]
mat_mnist   <- as.matrix(df_sampled[,!(colnames(df_sampled) == "Class")])
whichNumber <- as.character(df_sampled[,(colnames(df_sampled) == "Class")])

# Going to do parallel t-SNE later (and data has to be created for each instance on Windows R parallel,  so lets save the data as rds
saveRDS(mat_mnist,"data/mat_mnist.rds")
saveRDS(df_sampled,"data/df_sampled.rds")
saveRDS(whichNumber,"data/whichNumber.rds")

Run Parallel t-SNE

# Setup parallel
numCores <- parallel::detectCores()
numCores <- parallel::makeCluster(numCores-1)

# Run parallel t-SNE        
e <- parallel::parLapply(numCores, 1:30 , function(x){
          set.seed(42)
          # Read in data tyo each instance
          df_sampled <- readRDS("data/df_sampled.rds")          
          mat_mnist <- readRDS("data/mat_mnist.rds")
          whichNumber <- readRDS("data/whichNumber.rds")

          c<- Rtsne::Rtsne(df_sampled, pca=TRUE, dims=2, perplexity = x, theta = 0.5, max_iter = 1000)
          d <- as.data.frame(c$Y)
          d <- cbind.data.frame(whichNumber,d,rep(x,times=nrow(d)))
          colnames(d) <- c("Number","Dim.1","Dim.2","Framer")
          d
  
})
            parallel::stopCluster(numCores)

Create animation states and plots

tf <- tween_states(e, tweenlength = 2, statelength = 1, 
                   ease = c('cubic-in-out'), 
                   nframes = 500)


p <- ggplot(tf, aes(`Dim.1`, `Dim.2`, frame = Framer)) +
           geom_point(aes(colour=Number)) +
           ggtitle("t-SNE: Perplexity Step 1 to 30 by 1")

Create animation (GIF)

animation::ani.options(convert="C:\\ImageMagick-7.0.7-Q16\\convert.exe")
animation::ani.options(interval = 1/15)

gganimate(p,title_frame = FALSE)
