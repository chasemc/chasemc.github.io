rawData %>% group_by(year) %>% summarise_all(~sum(is.na(.))) %>% .[, 2:11] %>% max
## [1] 12

Find Max and Min values per year per field position

maxData <- rawData %>% group_by(year) %>% summarise_all(~max(.[!is.na(.)]))
minData <- rawData %>% group_by(year) %>% summarise_all(~min(.[!is.na(.)]))
b <- rawData %>% tidyr::gather(`Field Position`, Salary, -year)
b %<>% drop_na(Salary)

Let’s take look at the distribution of the data

ggplot(b) +
    facet_wrap(~year)+
    geom_violin(aes(y = Salary, x = `Field Position`))+
    theme(axis.text.x=element_blank()) +
    xlab("Years") +
    ylab("Salary")

ggplot(b) + 
    facet_wrap(~`Field Position`)+
    geom_boxplot(aes(y = Salary, x = as.factor(year)))+
    theme(axis.text.x=element_blank()) +
    xlab("Years") +
    ylab("Salary")

c <- maxData %>% tidyr::gather(`Field Position`, Salary, -year)
c %<>% drop_na(Salary)
d <- minData %>% tidyr::gather(`Field Position`, Salary, -year)
d %<>% drop_na(Salary)

colnames(b) <- c("Year", "Field Position", "Salary")
colnames(c) <- c("Year", "Field Position", "Salary")
colnames(d) <- c("Year", "Field Position", "Salary")
ggplot(c) + 
    facet_grid(~`Field Position`)+
    geom_line(data=c, aes(x=`Year`, y=`Salary`, colour=`Field Position`), lwd=1.5, alpha = .55)+
    geom_line(data=d, aes(x=`Year`, y=`Salary`, colour=`Field Position`), lwd=1.5, alpha = .55)+
    geom_jitter(data=b, aes(x=`Year`, y=`Salary`, colour=`Field Position`), lwd=1.5, alpha = .15)+
    scale_colour_discrete(guide = FALSE) +
    scale_y_continuous(labels=function(n){format(n, scientific = FALSE)})+
    theme(axis.text.x = element_text(angle = 90, vjust = .5))+
    labs(title= "Summary Plot os Salaries: \n Max (Lines), Min (Lines), All Values (Points)")+
    theme(plot.title=element_text(hjust=0.5))   

kermit <- rawData %>% group_by(year) %>% summarise_all(~moments::skewness(.[!is.na(.)]))

kermit<- kermit %>% tidyr::gather(`Field Position`, Salary, -year)


kermit %<>% drop_na(Salary)


ggplot(kermit) + 
    facet_grid(~`Field Position`)+
    geom_point(aes(x=year, y=Salary, colour=`Field Position`))+
    geom_line(aes(x=year, y=Salary, colour=`Field Position`), lwd=1.5, alpha=.5)+  
    theme(axis.text.x = element_text(angle = 90, vjust = .5))+
    xlab("Years") +
    ylab("Salary Skew")+
    labs(title= "Stalled Salaries Aren't the Only Issue for Running Backs, \n Skewed Salaries Persist as Well")+
    theme(plot.title=element_text(hjust=0.5))     

kermit <- rawData %>% group_by(year) %>% summarise_all(~moments::kurtosis(.[!is.na(.)]))

kermit<- kermit %>% tidyr::gather(`Field Position`, Salary, -year)


kermit %<>% drop_na(Salary)

st <- rawData %>% group_by(year) %>% summarise_all(~sd(.[!is.na(.)]))
st <- st %>% tidyr::gather(`Field Position`, Salary, -year)
st %<>% drop_na(Salary)

ggplot(st) +

    labs(title="Increasing Salary Standard Deviations Over Time")+
