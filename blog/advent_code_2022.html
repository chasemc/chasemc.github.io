prompt_data = "/home/chase/Downloads/input.txt"

current_calories = 0
cal_list = []
with open( "/home/chase/Downloads/input.txt") as fp:
    for line in fp:
        if line == "\n":
            cal_list.append(current_calories)
            current_calories = 0
        else:
            current_calories += int(line)
top_three = sorted(cal_list, reverse=True)[0:3]
print(sum(top_three))
