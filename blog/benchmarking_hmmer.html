Because hmmsearch can be IO limiting, it wasnt possible (at least I dont think) to just loop through conditions in nextflow so that the hmmsearch commands would run sequentially. Because of this the following bash script was used to inititate the nextflow workflow with varying the number of files the hmm models were split into.

#!/bin/bash
for i in {1..50}
do
nextflow run . \
--enable_conda true \
--outdir_per_run "/home/chase/Documents/test/per" \
--outdir_neo4j "/home/chase/Documents/test/neo" \
--outdir_long_cache "/home/chase/temp/socialgene/outdir_long_cache" -resume \
--numfiles "$i"
done

nextflow run . \
--enable_conda true \
--outdir_per_run "/home/chase/Documents/test/per" \
--outdir_neo4j "/home/chase/Documents/test/neo" \
--outdir_long_cache "/home/chase/temp/socialgene/outdir_long_cache" -resume \
--numfiles 100

nextflow run . \
--enable_conda true \
--outdir_per_run "/home/chase/Documents/test/per" \
--outdir_neo4j "/home/chase/Documents/test/neo" \
--outdir_long_cache "/home/chase/temp/socialgene/outdir_long_cache" -resume \
--numfiles 500

nextflow run . \
--enable_conda true \
--outdir_per_run "/home/chase/Documents/test/per" \
--outdir_neo4j "/home/chase/Documents/test/neo" \
--outdir_long_cache "/home/chase/temp/socialgene/outdir_long_cache" -resume \
--numfiles 1000
library(here)
## here() starts at /home/chase/Documents/github/socialgene/notebooks
library(data.table)
library(ggplot2)
library(lubridate)

##     yday, year

## 
##     date, intersect, setdiff, union
library(patchwork)
filepaths <- list.files(path = here::here("data","benchmarking_hmmer", "varying_hmm_split"),
                        pattern = "execution_trace*",
                        full.names = TRUE)

execution_traces <- lapply(filepaths, fread)

# extract the numfile variable (how many hmm file splits there were)
for (x in execution_traces) { 
  
  numfiles <- x[grep("BENCHMARK:HMM_HASH",name)]$name
  numfiles <- gsub("BENCHMARK:HMM_HASH (numfiles: ", "",  numfiles, fixed = TRUE)
  # assign new column that contains the number of hmm files 
  x[, numfiles := as.integer(gsub(")", "",  numfiles, fixed = TRUE))]
  
}
# combine into a single table
execution_traces <- do.call(rbind, execution_traces)

# filter for correct nextflowprocess
execution_traces <- execution_traces[grep("BENCHMARK:HMMSEARCH*", name )]
execution_traces <- execution_traces[status != "ABORTED"]
temp <- strsplit(execution_traces$duration, " ")
duration <- lapply(temp, 
                   function(x){
                     time_unit <- sub("[^a-z]+", "", x)
                     sum(as.vector(c(s = 1, m = 60)[time_unit]) * as.numeric(sub("[a-z]+$", "", x)))
                   })
execution_traces$duration_seconds <- unlist(duration)
plot_data <- execution_traces[, list(mean_duration =  mean(duration_seconds) / 60,
                        median_duration =  median(duration_seconds) / 60,
                        max_duration =  max(duration_seconds) / 60), numfiles]

plot_data <- data.table::melt(plot_data, id.vars=c("numfiles"), measure.vars=c("mean_duration",   "median_duration", "max_duration"))

ggplot(plot_data,aes(x = numfiles, value)) +
  geom_point() + 
  geom_line() + 
  ylab("Minutes") +
  facet_wrap(~variable)

p1 <- ggplot(plot_data[variable == "mean_duration" ],aes(x = numfiles, value)) +
    geom_point() +
